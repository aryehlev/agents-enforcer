Quick test — allow an agent to reach api.openai.com:443 only:

  cat <<EOF | kubectl apply -f -
  apiVersion: agents.enforcer.io/v1alpha1
  kind: GatewayCatalog
  metadata:
    name: platform
  spec:
    gateways:
      - { name: openai, host: 1.2.3.4, ports: [443] }  # <-- resolve first!
  ---
  apiVersion: agents.enforcer.io/v1alpha1
  kind: AgentPolicy
  metadata:
    name: openai-only
    namespace: default
  spec:
    podSelector:
      matchLabels: { app: ai-agent }
    egress:
      defaultAction: Deny
      gatewayRefs: [ openai ]
  EOF

Watch enforcement:
  kubectl get agentpolicies -A
  kubectl describe agentpolicy openai-only
